    <script>
        async function generate() {
            const notes = document.getElementById('notes').value;
            const num_cards = document.getElementById('num_cards').value;

            if (!notes.trim()) {
                alert('Please paste some notes first!');
                return;
            }

            document.getElementById('status').innerText = 'Generating... please wait';
            document.getElementById('cards').innerHTML = '';

            const response = await fetch('/generate', {
                method: 'POST',
                headers: { 'Content-Type': 'application/json' },
                body: JSON.stringify({ notes, num_cards })
            });

            const cards = await response.json();
            document.getElementById('status').innerText = `Generated ${cards.length} flashcards!`;
	    document.getElementById('download-btn').style.display = 'inline-block';
window.generatedCards = cards;
	async function download() {
    const response = await fetch('/download', {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({ cards: window.generatedCards })
    });
    const blob = await response.blob();
    const url = URL.createObjectURL(blob);
    const a = document.createElement('a');
    a.href = url;
    a.download = 'flashcards.csv';
    a.click();
}

            cards.forEach((card, i) => {
                document.getElementById('cards').innerHTML += `
                    <div class="card">
                        <p class="label">Q ${i + 1}</p>
                        <p>${card.question}</p>
                        <p class="label">A</p>
                        <p>${card.answer}</p>
                    </div>
                `;
            });
        }
    </script>
